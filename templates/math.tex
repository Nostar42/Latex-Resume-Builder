\begin{center}
  {\large\bfseries Problem}
\end{center}

Solve the quadratic equation:
\[
  2x^2 + 5x - 3 = 0
\]

\bigskip
\noindent\textbf{Step 1 --- Identify coefficients.}

We have $a = 2$, $b = 5$, $c = -3$.

\bigskip
\noindent\textbf{Step 2 --- Apply the quadratic formula.}

\[
  x = \frac{-b \pm \sqrt{b^2 - 4ac}}{2a}
    = \frac{-5 \pm \sqrt{25 + 24}}{4}
    = \frac{-5 \pm \sqrt{49}}{4}
    = \frac{-5 \pm 7}{4}
\]

\bigskip
\noindent\textbf{Step 3 --- Evaluate both roots.}

\begin{align*}
  x_1 &= \frac{-5 + 7}{4} = \frac{2}{4} = \frac{1}{2} \\[4pt]
  x_2 &= \frac{-5 - 7}{4} = \frac{-12}{4} = -3
\end{align*}

\bigskip
\noindent\textbf{Answer:}

\[
  \boxed{x = \tfrac{1}{2} \quad \text{or} \quad x = -3}
\]
